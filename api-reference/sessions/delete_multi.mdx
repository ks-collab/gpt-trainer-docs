<ParamField body="uuids" type="string[]" required/>

<RequestExample>

```bash Curl
curl --location --request POST 'https://app.gpt-trainer.com/api/v1/sessions/delete' \
--header 'Content-Type: application/json' \
--header 'Authorization: Bearer <token>'
--data-raw '{
  "uuids": []
}'
```

```py Python
import requests

url = 'https://app.gpt-trainer.com/api/v1/sessions/delete'
headers = {
    'Content-Type': 'application/json',
    'Authorization': 'Bearer <token>'
}

data = {
    "uuids": [],
}

response = requests.POST(url, headers=headers, json=data)

if response.status_code == 200:
    print("Request successful! Sessions deleted.")
else:
    print("Request failed with status code:", response.status_code)
    print(response.text)
```

```ts JavaScript
const axios = require('axios');

const url = 'https://app.gpt-trainer.com/api/v1/sessions/delete';
const headers = {
    'Content-Type': 'application/json',
    'Authorization': 'Bearer <token>'
};;

const data = {
    "uuids": []
};

axios.post(url, data, { headers })
    .then(response => {
        console.log('Request successful! Sessions deleted.');
    })
    .catch(error => {
        console.error('Request failed:', error);
    });
```
